1. Создать базу данных personal.

    mysql> create database personal;
    mysql> use personal;

2. Создать таблицу workers с полями: id, name, age, salary, где
- id - int 5, primary key, autoincrement
- name - varchar 50
- age - int 2
- salary - 5
Кодировка по умолчанию utf-8.

    mysql> create table workers ( id int(5) primary key auto_increment, name varchar(50) , age int(2), salary int(5) ) charset utf8;

3. Заполнить таблицу данными представленными на рисунке.

    mysql> insert into workers values (1,'Дима',23,400);
    mysql> insert into workers values (2,'Петя',25,500);
    mysql> insert into workers values (3,'Вася',23,500);
    mysql> insert into workers values (4,'Коля',30,1000);
    mysql> insert into workers values (5,'Иван',20,500);
    mysql> insert into workers values (6,'Кирилл',28,1000);

4. Выбрать работника с id = 2

    mysql> select name from workers where id=2;

5. Выбрать работников с зарплатой 500$

    mysql> select name from workers where salary=500;

6. Выбрать работников с зарплатой 500$ и id больше 3

    mysql> select name from workers where salary=500 and id>3;

7. Добавить нового работника Джона, 20 лет, зарплата 700$

    mysql> insert into workers (name, age, salary) values ('Джон',20,700);

8. Добавьте одним запросом трех новых работников: Катю, 20 лет, зарплата 500$,
Юлю, 25 лет, зарплата 600$, Женю, 30 лет, зарплата 900$.

    mysql> insert into workers (name, age, salary) values (
        -> 'Катя',20,500),(

9. Удалите работника Джона.

    mysql> delete from workers where name='Джон';

10. Поставьте Диме зарплату в 900$.

    mysql> update workers set salary=900 where name='Дима';

11. Поставьте Диме зарплату в 1000$ и возраст 20 лет.

    mysql> update workers set salary=1000, age=20  where name='Дима';

12. Выбрать работников в возрасте от 25 (не включительно) до 28 лет
(включительно).

    mysql> select name from workers where age>25 and age<=28;

13. Выбрать всех работников в возрасте 27 лет или с зарплатой 1000$

    mysql> select name from workers where age=27 or salary=1000;

14. Выбрать всех работников в возрасте от 23 лет до 27 лет или с зарплатой от
400$ до 1000$

    mysql> select name from workers where age between 23 and 27 or salary between 400 and 1000;
